cmake_minimum_required(VERSION 3.20)
project(nltools)

SET(CPACK_PACKAGE_VERSION_PATCH ${PROJECT_VERSION_PATCH})
SET(CPACK_DEBIAN_PACKAGE_DEPENDS "")
SET(CPACK_DEBIAN_PACKAGE_DESCRIPTION "Tools library for Nonlinear Labs software")
INCLUDE(CPack)
