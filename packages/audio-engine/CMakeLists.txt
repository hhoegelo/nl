cmake_minimum_required(VERSION 3.20)
project(audio-engine VERSION 1.0.1)

add_executable(${PROJECT_NAME} main.cpp)

SET(CPACK_GENERATOR "DEB")
SET(CPACK_BINARY_DEB On)
SET(CPACK_PACKAGE_CONTACT "info@nonlinear-labs.de")
SET(CPACK_DEBIAN_PACKAGE_MAINTAINER "Nonlinear Labs GmbH")
SET(CPACK_PACKAGE_VERSION_MAJOR ${PROJECT_VERSION_MAJOR})
SET(CPACK_PACKAGE_VERSION_MINOR ${PROJECT_VERSION_MINOR})
SET(CPACK_PACKAGE_VERSION_PATCH ${PROJECT_VERSION_PATCH})
SET(CPACK_DEBIAN_PACKAGE_DEPENDS "")
SET(CPACK_DEBIAN_PACKAGE_DESCRIPTION "Audio Engine for Nonlinear Labs synthesizers")
INCLUDE(CPack)
